import sys
import threading
from pathlib import Path
from importlib import import_module

# ---------------------------------------------------------------------------
# Correct PySide6 imports – QShortcut and QKeySequence are in QtGui
# ---------------------------------------------------------------------------
from PySide6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QHBoxLayout,
    QTextEdit, QLineEdit, QPushButton, QLabel
)
from PySide6.QtGui import QIcon, QTextCursor, QShortcut, QKeySequence
from PySide6.QtCore import Qt, Signal, QObject

# ---------------------------------------------------------------------------
# Ensure project root is on sys.path and import the OSGPT core module
# ---------------------------------------------------------------------------
sys.path.append(str(Path(__file__).resolve().parents[1]))
core = import_module('main')  # core.run_task will be used
import memory

# ---------------------------------------------------------------------------
# Worker thread and signals for background execution
# ---------------------------------------------------------------------------
class WorkerSignals(QObject):
    result = Signal(str)
    finished = Signal()

class TaskThread(threading.Thread):
    def __init__(self, request: str, signals: WorkerSignals):
        super().__init__(daemon=True)
        self.request = request
        self.signals = signals

    def run(self):
        # Build the system message exactly as the original entry point does
        from config import SYSTEM_PROMPT
        from ai import env_summary
        system_content = SYSTEM_PROMPT + "\n\nMachine info:\n" + env_summary()
        notes = memory.load_notes()
        if notes:
            system_content += "\n\nLong‑term memory (things you were told to remember):\n" + notes
        messages = [{"role": "system", "content": system_content}]

        # Capture stdout while OSGPT processes the request
        from io import StringIO
        old_stdout = sys.stdout
        sys.stdout = StringIO()
        try:
            core.run_task(self.request, messages, auto_approve=True)
            raw_output = sys.stdout.getvalue()
        finally:
            sys.stdout = old_stdout

        # Strip internal debug lines and any fallback notices
        cleaned = []
        for line in raw_output.splitlines():
            stripped = line.strip()
            if stripped.startswith("[command]") or stripped.startswith("[thought]") or stripped.startswith("[result]"):
                continue
            if "[fallback]" in stripped:
                continue  # hide rate‑limit fallback messages
            cleaned.append(line)
        output = "\n".join(cleaned).strip()
        self.signals.result.emit(output)
        self.signals.finished.emit()

# ---------------------------------------------------------------------------
# Main Window – neon hacker theme
# ---------------------------------------------------------------------------
class OSGPTWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("🧠 OSGPT – Neon Hacker Assistant")
        self.resize(960, 640)
        self.setWindowIcon(QIcon())
        self._setup_ui()
        self._setup_shortcuts()

    def _setup_ui(self):
        self.setStyleSheet(self._global_style())
        layout = QVBoxLayout(self)
        layout.setContentsMargins(12, 12, 12, 12)
        layout.setSpacing(8)

        self.chat_view = QTextEdit(self)
        self.chat_view.setReadOnly(True)
        self.chat_view.setAcceptRichText(True)
        self.chat_view.setStyleSheet("background:transparent; border:none;")
        layout.addWidget(self.chat_view, 1)

        self.thinking = QLabel("🟢 thinking...", self)
        self.thinking.setStyleSheet("color:#00ff00; font-family:Consolas; font-size:13px;")
        self.thinking.hide()
        layout.addWidget(self.thinking)

        input_layout = QHBoxLayout()
        self.input_box = QLineEdit(self)
        self.input_box.setPlaceholderText("Enter your request…")
        self.input_box.setStyleSheet("padding:6px; border-radius:4px; background:#0a0a0a; color:#00ff00;")
        self.send_btn = QPushButton("Send", self)
        self.send_btn.setCursor(Qt.PointingHandCursor)
        self.send_btn.setStyleSheet(self._button_style())
        self.send_btn.clicked.connect(self._on_send)
        input_layout.addWidget(self.input_box, 1)
        input_layout.addWidget(self.send_btn)
        layout.addLayout(input_layout)

    def _global_style(self) -> str:
        return (
            "QWidget { background:#0a0a0a; color:#00ff00; font-family:Consolas, 'Segoe UI', sans-serif; }"
            "QScrollBar:vertical { width:12px; background:#111111; }"
            "QScrollBar::handle:vertical { background:#222222; border-radius:4px; }"
        )

    def _button_style(self) -> str:
        return (
            "QPushButton { background:#00ff00; color:#000000; border:none; padding:6px 12px; border-radius:4px; }"
            "QPushButton:hover { background:#33ff33; }"
            "QPushButton:pressed { background:#009900; }"
        )

    def _setup_shortcuts(self):
        QShortcut(QKeySequence("Ctrl+Return"), self, activated=self._on_send)
        QShortcut(QKeySequence("Ctrl+L"), self, activated=self._clear_chat)

    def _append_html(self, html: str):
        self.chat_view.moveCursor(QTextCursor.End)
        self.chat_view.insertHtml(html + "<br>")
        self.chat_view.moveCursor(QTextCursor.End)
        self.chat_view.viewport().update()

    def _on_send(self):
        request = self.input_box.text().strip()
        if not request:
            return
        self._append_html(f"<span style='color:#00ffff'>you&gt; {request}</span>")
        self.input_box.clear()
        self.send_btn.setEnabled(False)
        self.thinking.show()
        signals = WorkerSignals()
        signals.result.connect(self._display_result)
        signals.finished.connect(self._task_done)
        TaskThread(request, signals).start()

    def _display_result(self, output: str):
        if not output:
            output = "(no visible output)"
        # Ensure any stray fallback markers are removed (already filtered, but double‑check)
        safe_output = "\n".join([ln for ln in output.splitlines() if "[fallback]" not in ln])
        formatted = (
            f"<div style='margin-top:4px; margin-bottom:8px;'>"
            f"<span style='color:#ff00ff'>OSGPT&gt; </span>"
            f"<span style='color:#ffcc00'>{safe_output.replace('\n', '<br>')}</span>"
            f"</div>"
        )
        self._append_html(formatted)

    def _task_done(self):
        self.thinking.hide()
        self.send_btn.setEnabled(True)
        self.chat_view.verticalScrollBar().setValue(self.chat_view.verticalScrollBar().maximum())

    def _clear_chat(self):
        self.chat_view.clear()
        self._append_html("<i>Chat cleared.</i>")

def launch():
    app = QApplication(sys.argv)
    app.setAttribute(Qt.AA_EnableHighDpiScaling)
    win = OSGPTWindow()
    win.show()
    sys.exit(app.exec())

if __name__ == "__main__":
    launch()
